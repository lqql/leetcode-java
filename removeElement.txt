public class Solution {
    public int removeElement(int[] nums, int val) {
        if(nums==null||nums.length==0)
            return 0;
        int i;
        int j=0;
        for(i=0;i<nums.length;i++){
            if(nums[i]==val)
                continue;
            nums[j]=nums[i];
            j++;
        }
        return j;
    }
}
